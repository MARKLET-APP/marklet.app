4 ADD UNIVERSAL SHARE BUTTON
================================================

Create file:

/public/js/shareListing.js

Code:

function shareListing(listing){

const url = window.location.origin + "/listing/" + listing.id

const text =
`${listing.title}

${listing.city || ""}

شاهد الإعلان على MARKLET:
${url}`

if(navigator.share){

navigator.share({
title: listing.title,
text: text,
url: url
})

}else{

navigator.clipboard.writeText(text)

alert("تم نسخ رابط الإعلان للمشاركة")

}

}
